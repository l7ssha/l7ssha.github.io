Future<Null> send(
    {String content,
    List<Map<String, dynamic>> embeds,
    String username,
    String avatarUrl,
    bool tts}) async {
  Map<String, dynamic> payload = {
    "content": content,
    "username": username,
    "avatar_url": avatarUrl,
    "tts": tts,
    "embeds": embeds
  };

  await this.http.send('POST', "/webhooks/$id/$token", body: payload);
  return null;
}
