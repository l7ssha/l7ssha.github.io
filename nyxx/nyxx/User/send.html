@override

/// Sends a message.
Future<Message> send(
    {Object content = "",
    EmbedBuilder embed,
    bool tts = false,
    bool disableEveryone}) async {
  DMChannel channel = await getDMChannel();
  return await channel.send(
      content: content,
      embed: embed,
      tts: tts,
      disableEveryone: disableEveryone);
}
