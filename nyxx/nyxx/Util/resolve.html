String resolve (String to, object)

Resolves an object into a target object.

Implementation 

static String resolve(String to, dynamic object) {
  if (to == "channel") {
    if (object is Message) {
      return object.channel.id;
    } else if (object is Channel) {
      return object.id;
    } else if (object is Guild) {
      return object.defaultChannel.id;
    } else {
      return object.toString();
    }
  } else if (to == "message") {
    if (object is Message) {
      return object.id;
    } else {
      return object.toString();
    }
  } else if (to == "guild") {
    if (object is Message) {
      return object.guild.id;
    } else if (object is GuildChannel) {
      return object.guild.id;
    } else if (object is Guild) {
      return object.id;
    } else {
      return object.toString();
    }
  } else if (to == "user") {
    if (object is Message) {
      return object.author.id;
    } else if (object is User) {
      return object.id;
    } else if (object is Member) {
      return object.id;
    } else {
      return object.toString();
    }
  } else if (to == "member") {
    if (object is Message) {
      return object.author.id;
    } else if (object is User) {
      return object.id;
    } else if (object is Member) {
      return object.id;
    } else {
      return object.toString();
    }
  } else if (to == "app") {
    if (object is User) {
      return object.id;
    } else if (object is Member) {
      return object.id;
    } else {
      return object.toString();
    }
  } else {
    return null;
  }
}
