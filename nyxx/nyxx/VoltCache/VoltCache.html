VoltCache(Duration duration, bool roundup(T key, S value, Timer t)) {
  _cache = Map();

  Timer.periodic(duration, (t) {
    _cache.forEach((k, v) {
      if(roundup(k, v, t))
        _cache.remove(k);
    });
  });
}
