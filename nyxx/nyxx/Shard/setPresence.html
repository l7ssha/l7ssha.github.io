void setPresence({String status, bool afk, Game game}) {
  Map<String, dynamic> packet = {
    "afk": afk,
    "since": null,
    "status": status,
    "game": {
      "name": game != null ? game.name : null,
      "type": game != null ? game.type : null,
      "url": game != null ? game.url : null
    }
  };

  this.send("STATUS_UPDATE", packet);
}
