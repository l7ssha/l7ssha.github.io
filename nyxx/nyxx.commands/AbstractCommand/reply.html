Future<Message> reply(
    {String content,
    EmbedBuilder embed,
    bool tts: false,
    String nonce,
    bool disableEveryone}) async {
  return await context.message.channel.send(
      content: content,
      embed: embed,
      tts: tts,
      nonce: nonce,
      disableEveryone: disableEveryone);
}
