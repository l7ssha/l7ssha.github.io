Future<Message> replyDelayed(Duration duration,
    {String content,
    EmbedBuilder embed,
    bool tts: false,
    String nonce,
    bool disableEveryone}) async {
  return new Future.delayed(
      duration,
      () async => await channel.send(
          content: content,
          embed: embed,
          tts: tts,
          nonce: nonce,
          disableEveryone: disableEveryone));
}
