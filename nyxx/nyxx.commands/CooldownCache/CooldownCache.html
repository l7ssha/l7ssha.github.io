CooldownCache() {
  _cache = [];

  new Timer.periodic(const Duration(minutes: 30), (Timer t) {
    int now = new DateTime.now().millisecondsSinceEpoch;

    _cache.removeWhere((item) => now - item.lastUsed >= 1800000);
  });
}
