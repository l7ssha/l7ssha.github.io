Future connect() async {
  var wc;
  for (var i = 6463; i < 6472; i++) {
    var path =
        Uri.parse("ws://127.0.0.1:$i/?v=1&client_id=$clientId&encoding=json");
    try {
      var c = await w_transport.WebSocket.connect(path, headers: {
        "RPC Origin": "https://localhost",
        "Redirect URI": "https://localhost"
      });
      wc = c;
      break;
    } catch (e) {
      print("FAILED at $i");
    }
  }

  print("CONNECTED!");
  print(wc);

  return new Future.delayed(const Duration(seconds: 2), () async {
    var authorize = {
      "args": {
        "client_id": clientId,
        "scopes": ["rpc.api", "rpc", "identify"]
      },
      "cmd": "AUTHORIZE"
    };

    wc.add(jsonEncode(authorize));
    print("SENT");

    var res = await wc.first as String;
    print("HAA");
    print(jsonDecode(res));
  });
}
