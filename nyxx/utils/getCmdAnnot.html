T getCmdAnnot <T>(DeclarationMirror declaration)

Gets single annotation with type T from declaration

T getCmdAnnot<T>(DeclarationMirror declaration) {
  Iterable<T> fs = getCmdAnnots<T>(declaration);
  if (fs.isEmpty) return null;
  return fs.first;
}
